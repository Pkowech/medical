#### 2. **NEW: WeaknessOrchestrationService**
**File:** `backend/src/modules/education/assessment/services/weakness-orchestration.service.ts`

```typescript
@Injectable()
export class WeaknessOrchestrationService {
  constructor(
    private weaknessChainService: WeaknessChainService,
    private quizGenerationService: QuizGenerationService,
    private assessmentProgressService: AssessmentProgressService,
    private notificationService: NotificationService,
    private prisma: PrismaService,
  ) {}

  /**
   * Main orchestration: Listen for assessment completion
   * Detect weakness → Auto-generate quiz → Notify user
   */
  @OnEvent('assessment.completed')
  async onAssessmentComplete(event: AssessmentCompleteEvent): Promise<void> {
    const { userId, courseId, topicId, score } = event;

    // 1. Run weakness detection for user
    const weaknessChainsdetected = await this.weaknessChainService.detectWeaknessChains(userId);

    if (weaknessChainsdetected.length === 0) return; // No weaknesses

    // 2. For each HIGH/MEDIUM weakness, generate quiz
    for (const chain of weaknessChainsdetected.filter(c => c.riskLevel !== 'LOW')) {
      const quiz = await this.quizGenerationService.generateQuizForWeakness(
        userId,
        chain,
        5, // 5 questions
      );

      // 3. Notify user
      await this.notificationService.sendNotification({
        userId,
        title: `AI Detected Weakness in ${chain.weakTopic.name}`,
        body: `We created a custom 5-question quiz to help strengthen this area.`,
        actionUrl: `/quiz/${quiz.id}`,
        priority: 'high',
      });

      // 4. Store relationship for tracking
      await this.prisma.autoGeneratedQuiz.create({
        data: {
          userId,
          quizId: quiz.id,
          weaknessTopicId: chain.weakTopic.id,
          detectedAt: new Date(),
          riskLevel: chain.riskLevel,
        },
      });
    }
  }

  /**
   * After user completes auto-generated quiz:
   * Track improvement in UserSkillState + send feedback
   */
  @OnEvent('auto-quiz.completed')
  async onAutoQuizComplete(event: AutoQuizCompleteEvent): Promise<void> {
    const { userId, quizId, score, maxScore } = event;

    // 1. Fetch original weakness data
    const autoQuizRecord = await this.prisma.autoGeneratedQuiz.findUnique({
      where: { quizId },
    });

    if (!autoQuizRecord) return;

    // 2. Get old pKnown (before weakness was detected)
    const skillState = await this.prisma.userSkillState.findUnique({
      where: {
        userId_skillId: {
          userId,
          skillId: autoQuizRecord.weaknessTopicId,
        },
      },
    });

    const oldPKnown = skillState?.pKnown ?? 0;
    const newPKnown = Math.min(oldPKnown + (score / maxScore) * 0.2, 1.0);

    // 3. Send feedback
    const improvement = ((newPKnown - oldPKnown) / oldPKnown) * 100;
    await this.notificationService.sendNotification({
      userId,
      title: 'Great Progress!',
      body: `Your mastery improved from ${(oldPKnown * 100).toFixed(0)}% to ${(newPKnown * 100).toFixed(0)}%. Keep going!`,
      actionUrl: `/progress`,
      priority: 'high',
    });
  }
}
```

**Why This Service:**
- Decouples event handling from business logic
- Orchestrates multiple services (weakness, quiz, notification)
- Tracks complete workflow

@Injectable()
export class GroundedTutorService {
  constructor(
    private curriculumRAGService: CurriculumRAGService,
    private skillStateService: UserSkillStateService,
    private claudeService: ClaudeService,
  ) {}

  /**
   * Generate tutor response grounded in curriculum + student context
   */
  async generateGroundedResponse(
    userId: string,
    query: string,
    conversationHistory: Message[],
  ): Promise<{ response: string; citations: Citation[] }> {
    // 1. Retrieve curriculum context
    const curriculumContext = await this.curriculumRAGService.retrieveContext(
      query,
      userId,
    );

    // 2. Retrieve student context (weak areas, completed materials)
    const studentContext = await this.getStudentContext(userId);

    // 3. Build prompt with all context
    const prompt = this.buildTutorPrompt(
      query,
      curriculumContext,
      studentContext,
      conversationHistory,
    );

    // 4. Call Claude
    const response = await this.claudeService.chat({
      messages: [{ role: 'user', content: prompt }],
      model: 'claude-3-sonnet', // Or latest
    });

    // 5. Extract citations from curriculum context
    const citations = curriculumContext.map(c => ({
      title: c.materialTitle,
      source: c.materialId,
      relevance: c.score,
    }));

    return {
      response: response.content,
      citations,
    };
  }

@Injectable()
export class ExamSessionService {
  constructor(
    private prisma: PrismaService,
    private quizService: QuizService,
  ) {}

  /**
   * Initialize exam session
   * Locks all review/navigation features
   */
  async startExamSession(
    userId: string,
    examTemplateId: string,
    courseId: string,
  ): Promise<ExamSession> {
    // 1. Create exam session record
    const session = await this.prisma.examSession.create({
      data: {
        userId,
        examTemplateId,
        courseId,
        startedAt: new Date(),
        endTime: new Date(Date.now() + 90 * 60 * 1000), // 90 minutes
        status: 'active',
        responses: [],
      },
    });

    // 2. Generate question pool based on blueprint
    const questions = await this.generateExamQuestions(examTemplateId);

    // 3. Return session (DO NOT return all questions—return one at a time)
    return {
      sessionId: session.id,
      totalQuestions: questions.length,
      timeRemainingMinutes: 90,
      currentQuestion: questions[0],
      questionNumber: 1,
    };
  }

  /**
   * Submit answer and advance to next question
   */
  async submitAnswer(
    sessionId: string,
    questionId: string,
    answer: string,
    timeSpent: number, // seconds
  ): Promise<{
    isCorrect: boolean;
    nextQuestion?: Question;
    questionNumber: number;
    sessionComplete: boolean;
  }> {
    // 1. Validate answer
    const isCorrect = await this.validateAnswer(questionId, answer);

    // 2. Store response
    await this.prisma.examResponse.create({
      data: {
        sessionId,
        questionId,
        answer,
        isCorrect,
        timeSpent,
      },
    });

    // 3. Get next question
    const session = await this.prisma.examSession.findUnique({
      where: { id: sessionId },
    });

    const responseCount = await this.prisma.examResponse.count({
      where: { sessionId },
    });

    if (responseCount >= session.totalQuestions) {
      // Exam complete
      return { isCorrect, questionNumber: responseCount, sessionComplete: true };
    }

    // 4. Return next question
    const questions = await this.prisma.question.findMany(
      // ... exam session questions
    );
    const nextQuestion = questions[responseCount];

    return {
      isCorrect,
      nextQuestion,
      questionNumber: responseCount + 1,
      sessionComplete: false,
    };
  }

  /**
   * Finalize exam and generate report
   */
  async finalizeExam(sessionId: string): Promise<ExamReport> {
    // 1. Calculate score
    const responses = await this.prisma.examResponse.findMany({
      where: { sessionId },
    });

    const correctCount = responses.filter(r => r.isCorrect).length;
    const rawScore = (correctCount / responses.length) * 100;

    // 2. Calculate percentile (vs. other students)
    const percentile = await this.calculatePercentile(rawScore);

    // 3. Identify strong/weak areas
    const byTopic = groupBy(responses, 'question.topicId');
    const strengths = Object.entries(byTopic)
      .filter(([_, rs]) => (rs.filter(r => r.isCorrect).length / rs.length) > 0.75)
      .map(([topicId, _]) => topicId);
    const weaknesses = Object.entries(byTopic)
      .filter(([_, rs]) => (rs.filter(r => r.isCorrect).length / rs.length) < 0.50)
      .map(([topicId, _]) => topicId);

    // 4. Time analysis
    const avgTimePerQuestion = responses.reduce((a, r) => a + r.timeSpent, 0) / responses.length;

    return {
      score: rawScore,
      percentile,
      strengths,
      weaknesses,
      averageTimePerQuestion: avgTimePerQuestion,
      totalTime: responses.reduce((a, r) => a + r.timeSpent, 0),
    };
  }
}
